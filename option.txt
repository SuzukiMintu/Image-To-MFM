### 読み込む画像ファイルのパス
filename = aaaa.png

### ノート用かページ用かの設定
### 0: ノート
### 1: ページ
note_or_page = 0

### 画像のリサイズ指定
### どちらかのみ設定(片方を0などの無効な値に)するともう片方は自動計算される
resize_width = 14
resize_height = 0

### 減色値
### 1.0以上の小数値で指定。設定した値で色を割り算して減色する
### 1.0なら減色なし、2.0なら2分の1に減色
### 推奨値 32.0
color_division = 32.0

### カラーの設定
### 0: 6桁RGB
### 1: 3桁RGB
### 2: 4桁RGBA
color_type = 1
